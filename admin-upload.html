/** * Nexus Pro 2.0 - Core Administrative Library Publication Controller * File: admin-upload.js */ const DB_BASE_URL = "https://nexuspro-cf948-default-rtdb.europe-west1.firebasedatabase.app"; async function verifyAdminCredentials() { const userId = localStorage.getItem('nexusUserId'); const secureToken = localStorage.getItem('nexusAuthToken'); if (!userId || !secureToken) { window.location.replace('login.html'); return null; } try { // Double check permissions directly with the profile configuration schema const response = await fetch(`${DB_BASE_URL}/users/${userId}.json?auth=${secureToken}`); if (!response.ok) throw new Error("Security verification communication fault."); const userData = await response.json(); if (!userData || (userData.role || '').toLowerCase().trim() !== 'admin') { alert("🔒 Access violation detected. Administrative credentials required."); window.location.replace('dashboard.html'); return null; } return secureToken; // Valid session confirmed } catch (err) { console.error("Authorization check failure:", err.message); window.location.replace('dashboard.html'); return null; } } document.addEventListener("DOMContentLoaded", async () => { // Run verification immediately on entry const confirmedAuthToken = await verifyAdminCredentials(); if (!confirmedAuthToken) return; const uploadForm = document.getElementById('uploadForm'); const submitBtn = document.getElementById('submitBtn'); const btnText = document.getElementById('btnText'); uploadForm?.addEventListener('submit', async (e) => { e.preventDefault(); // Assemble structural data package const payload = { title: document.getElementById('materialTitle').value.trim(), subject: document.getElementById('materialSubject').value, category: document.getElementById('materialCategory').value, url: document.getElementById('materialLink').value.trim(), timestamp: Date.now() }; // UI Loading state update if (submitBtn) submitBtn.disabled = true; if (btnText) btnText.innerText = "Transmitting Resource..."; try { // Push directly to the /library list node endpoint const targetUrl = `${DB_BASE_URL}/library.json?auth=${confirmedAuthToken}`; const uploadResponse = await fetch(targetUrl, { method: 'POST', // POST generates an auto-incremented UUID token node inside Firebase RTDB headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); if (!uploadResponse.ok) throw new Error("Database refused transaction."); alert("✅ Material successfully compiled and added to the E-Library!"); uploadForm.reset(); } catch (error) { console.error("Upload failure tracker node:", error.message); alert(`⚠️ Upload rejected: ${error.message}`); } finally { // Restore button properties if (submitBtn) submitBtn.disabled = false; if (btnText) btnText.innerText = "Deploy to Library Catalog"; } }); });
